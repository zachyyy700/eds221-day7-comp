  group_by(sex) |>

str_detect()

my_string <- "Teddy loves eating salmon and socks."

# Does the pattern "love" exist within the string?
my_string |> 
  str_detect("love")
[1] TRUE
# also on vectors
my_string <- c("burrito", "fish taco", "Taco salad")

# Does the vector element contain the pattern "fish"?
my_string |> 
  str_detect("fish")
[1] FALSE  TRUE FALSE
